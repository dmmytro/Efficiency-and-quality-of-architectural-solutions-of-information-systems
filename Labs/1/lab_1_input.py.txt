# Вихідний код "довгого методу"

def calculate_area(length, width):
    # Перевірка наявності введених значень
    if length <= 0 or width <= 0:
        print("Помилка: Некоректні значення довжини або ширини")
        return
    
    # Обчислення площі
    area = length * width
    
    # Виведення результату
    print(f"Площа прямокутника з довжиною {length} та шириною {width} дорівнює {area}")

# Виклик функції для обчислення площі
calculate_area(5, 10)
